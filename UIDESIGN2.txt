local function Tween(obj, duration, props, style, dir)
    style = style or Enum.EasingStyle.Quart
    dir   = dir   or Enum.EasingDirection.Out
    local ti = TweenInfo.new(duration, style, dir)
    TweenSvc:Create(obj, ti, props):Play()
end

local function SpringTween(obj, duration, props)
    Tween(obj, duration, props, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
end

local function New(class, props, children)
    local inst = Instance.new(class)
    if props then
        for k, v in pairs(props) do
            if k ~= "Parent" then
                pcall(function() inst[k] = v end)
            end
        end
    end
    if children then
        for _, child in ipairs(children) do
            child.Parent = inst
        end
    end
    if props and props.Parent then
        inst.Parent = props.Parent
    end
    return inst
end

local function Corner(radius, parent)
    return New("UICorner", { CornerRadius = UDim.new(0, radius), Parent = parent })
end

local function Stroke(thickness, color, transparency, parent)
    return New("UIStroke", {
        Thickness     = thickness or 1,
        Color         = color or T.Outline,
        Transparency  = transparency or 0,
        Parent        = parent,
    })
end

local function Gradient(colorSeq, rotation, parent)
    return New("UIGradient", {
        Color    = colorSeq,
        Rotation = rotation or 0,
        Parent   = parent,
    })
end

local function Shadow(parent, size, transparency)
    return New("ImageLabel", {
        Name                 = "Shadow",
        AnchorPoint          = Vector2.new(0.5, 0.5),
        Position             = UDim2.new(0.5, 0, 0.5, 0),
        Size                 = UDim2.new(1, size or 30, 1, size or 30),
        BackgroundTransparency = 1,
        Image                = "rbxassetid://1316045217",
        ImageColor3          = Color3.new(0, 0, 0),
        ImageTransparency    = transparency or 0.4,
        ScaleType            = Enum.ScaleType.Slice,
        SliceCenter          = Rect.new(10, 10, 118, 118),
        ZIndex               = 0,
        Parent               = parent,
    })
end

local function LerpColor(a, b, t)
    return Color3.new(
        a.R + (b.R - a.R) * t,
        a.G + (b.G - a.G) * t,
        a.B + (b.B - a.B) * t
    )
end

-- ═══════════════════════════════════════════════════
-- 5. DRAG SYSTEM  (PC + Mobile, clamp to screen)
-- ═══════════════════════════════════════════════════
local function MakeDraggable(handle, target)
    local dragging, startMouse, startPos = false, nil, nil

    local function onInputBegan(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            dragging   = true
            startMouse = Vector2.new(input.Position.X, input.Position.Y)
            startPos   = target.Position
        end
    end

    local function onInputChanged(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement
            or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = Vector2.new(input.Position.X, input.Position.Y) - startMouse
            local vp    = workspace.CurrentCamera.ViewportSize
            local newX  = math.clamp(startPos.X.Offset + delta.X, 0, vp.X - target.AbsoluteSize.X)
            local newY  = math.clamp(startPos.Y.Offset + delta.Y, 0, vp.Y - target.AbsoluteSize.Y)
            target.Position = UDim2.fromOffset(newX, newY)
        end
    end

    local function onInputEnded(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end

    handle.InputBegan:Connect(onInputBegan)
    handle.InputChanged:Connect(onInputChanged)
    InputSvc.InputChanged:Connect(onInputChanged)
    InputSvc.InputEnded:Connect(onInputEnded)
end

local function AnimateOpen(frame, transparency)
    frame.Size = UDim2.new(frame.Size.X.Scale, frame.Size.X.Offset * 0.9,
                            frame.Size.Y.Scale, frame.Size.Y.Offset * 0.9)
    frame.GroupTransparency = 1
    frame.Visible = true
    SpringTween(frame, 0.3, {
        Size = UDim2.fromOffset(frame.Size.X.Offset / 0.9, frame.Size.Y.Offset / 0.9),
        GroupTransparency = transparency or 0,
    })
end

local function AnimateClose(frame, callback)
    Tween(frame, 0.2, {
        GroupTransparency = 1,
    }, Enum.EasingStyle.Quad, Enum.EasingDirection.In)
    task.delay(0.22, function()
        frame.Visible = false
        if callback then callback() end
    end)
end

-- Ripple on click inside a button
local function AddRipple(button)
    button.ClipsDescendants = true
    button.MouseButton1Down:Connect(function(mx, my)
        local rip = New("Frame", {
            Name                 = "Ripple",
            BackgroundColor3     = Color3.new(1,1,1),
            BackgroundTransparency = 0.85,
            AnchorPoint          = Vector2.new(0.5, 0.5),
            Position             = UDim2.fromOffset(
                mx - button.AbsolutePosition.X,
                my - button.AbsolutePosition.Y
            ),
            Size                 = UDim2.fromOffset(0, 0),
            ZIndex               = button.ZIndex + 10,
            Parent               = button,
        })
        Corner(100, rip)
        local target = math.max(button.AbsoluteSize.X, button.AbsoluteSize.Y) * 2.2
        Tween(rip, 0.5, {
            Size                 = UDim2.fromOffset(target, target),
            BackgroundTransparency = 1,
        }, Enum.EasingStyle.Quad)
        task.delay(0.55, function() rip:Destroy() end)
    end)
end

-- Hover shimmer for components
local function HoverEffect(frame, hoverColor, baseColor)
    hoverColor = hoverColor or LerpColor(frame.BackgroundColor3, Color3.new(1,1,1), 0.06)
    baseColor  = baseColor  or frame.BackgroundColor3
    frame.MouseEnter:Connect(function()
        Tween(frame, 0.18, { BackgroundColor3 = hoverColor })
    end)
    frame.MouseLeave:Connect(function()
        Tween(frame, 0.22, { BackgroundColor3 = baseColor })
    end)
end

-- Press scale for buttons
local function PressEffect(button)
    button.MouseButton1Down:Connect(function()
        Tween(button, 0.07, {
            Size = UDim2.new(
                button.Size.X.Scale, button.Size.X.Offset - 3,
                button.Size.Y.Scale, button.Size.Y.Offset - 2
            ),
        }, Enum.EasingStyle.Quad)
    end)
    local orig = button.Size
    button.MouseButton1Up:Connect(function()
        SpringTween(button, 0.18, { Size = orig })
    end)
    button.MouseLeave:Connect(function()
        SpringTween(button, 0.18, { Size = orig })
    end)
end

    end)

local function LoadFlags()
    pcall(function()
        local path = FOLDER .. "/" .. tostring(game.GameId) .. ".json"
        if isfile(path) then
            local ok, decoded = pcall(HttpSvc.JSONDecode, HttpSvc, readfile(path))
            if ok and type(decoded) == "table" then flags = decoded end
        end
    end)
end

-- 9. THEME APPLICATION HELPER

local ThemeCallbacks = {}  -- array of functions to call when theme changes

local function ApplyTheme(themeName)
    local preset = ThemePresets[themeName]
    if not preset then return end
    for k, v in pairs(preset) do T[k] = v end
    for _, cb in ipairs(ThemeCallbacks) do
        pcall(cb)
    end
end

local function OnThemeChange(cb)
    table.insert(ThemeCallbacks, cb)
end

        ---------- SLIDER ----------------------------------------
        function TabAPI:AddSlider(opts)
            opts = opts or {}
            local key     = opts.Title or "slider_" .. math.random(1000,9999)
            local minVal  = opts.Min      or 0
            local maxVal  = opts.Max      or 100
            local defVal  = opts.Default  or minVal
            local decAmt  = opts.Decimals or 0
            local cb      = opts.Callback or function() end
            local suffix  = opts.Suffix   or ""

            if flags[key] then
                defVal = math.clamp(flags[key], minVal, maxVal)
            end
            local sliderVal = defVal

            local card = New("Frame", {
                BackgroundColor3 = T.Component,
                Size             = UDim2.new(1, 0, 0, 58),
                LayoutOrder      = opts.Order or 0,
                Parent           = page,
            })
            Corner(10, card)
            Stroke(1, T.Outline, 0, card)

            New("TextLabel", {
                BackgroundTransparency = 1,
                Text             = opts.Title or "Slider",
                TextColor3       = T.Title,
                Font             = Enum.Font.GothamSemibold,
                TextSize         = 13,
                TextXAlignment   = Enum.TextXAlignment.Left,
                Size             = UDim2.new(1, -80, 0, 20),
                Position         = UDim2.new(0, 12, 0, 8),
                ZIndex           = card.ZIndex + 1,
                Parent           = card,
            })
            if opts.Description then
                New("TextLabel", {
                    BackgroundTransparency = 1,
                    Text             = opts.Description,
                    TextColor3       = T.Description,
                    Font             = Enum.Font.Gotham,
                    TextSize         = 10,
                    TextXAlignment   = Enum.TextXAlignment.Left,
                    Size             = UDim2.new(1, -80, 0, 14),
                    Position         = UDim2.new(0, 12, 0, 26),
                    ZIndex           = card.ZIndex + 1,
                    Parent           = card,
                })
            end

            local valLabel = New("TextLabel", {
                BackgroundTransparency = 1,
                TextColor3       = T.Accent,
                Font             = Enum.Font.GothamBold,
                TextSize         = 12,
                TextXAlignment   = Enum.TextXAlignment.Right,
                Size             = UDim2.new(0, 64, 0, 20),
                Position         = UDim2.new(1, -14, 0, 8),
                AnchorPoint      = Vector2.new(1, 0),
                ZIndex           = card.ZIndex + 1,
                Parent           = card,
            })

            -- Track
            local track = New("Frame", {
                BackgroundColor3 = T.Interactable,
                Size             = UDim2.new(1, -24, 0, 5),
                Position         = UDim2.new(0, 12, 1, -14),
                AnchorPoint      = Vector2.new(0, 1),
                ZIndex           = card.ZIndex + 1,
                ClipsDescendants = false,
                Parent           = card,
            })
            Corner(4, track)

            local fill = New("Frame", {
                BackgroundColor3 = T.Accent,
                Size             = UDim2.new(0, 0, 1, 0),
                ZIndex           = card.ZIndex + 2,
                ClipsDescendants = false,
                Parent           = track,
            })
            Corner(4, fill)

            -- Gradient on fill
            Gradient(ColorSequence.new({
                ColorSequenceKeypoint.new(0, T.AccentHover),
                ColorSequenceKeypoint.new(1, T.Accent),
            }), 0, fill)

            -- Thumb
            local thumb = New("Frame", {
                BackgroundColor3 = Color3.new(1,1,1),
                Size             = UDim2.fromOffset(14, 14),
                AnchorPoint      = Vector2.new(0.5, 0.5),
                Position         = UDim2.new(0, 0, 0.5, 0),
                ZIndex           = card.ZIndex + 3,
                Parent           = fill,
            })
            Corner(100, thumb)
            Stroke(2, T.Accent, 0.1, thumb)
            Shadow(thumb, 8, 0.55)

            local function FormatVal(v)
                local fmt = (decAmt > 0) and ("%." .. decAmt .. "f") or "%d"
                return string.format(fmt, v) .. suffix
            end

            local function UpdateSlider(scale)
                scale     = math.clamp(scale, 0, 1)
                sliderVal = minVal + (maxVal - minVal) * scale
                if decAmt == 0 then sliderVal = math.round(sliderVal) end
                valLabel.Text = FormatVal(sliderVal)
                Tween(fill, 0.08, { Size = UDim2.new(scale, 0, 1, 0) }, Enum.EasingStyle.Linear)
                thumb.Position = UDim2.new(1, 0, 0.5, 0)
                -- Note: flags/SaveFlags called only on InputEnded to avoid per-frame IO
                pcall(cb, sliderVal)
            end

            -- Init
            local initScale = (defVal - minVal) / (maxVal - minVal)
            fill.Size = UDim2.new(initScale, 0, 1, 0)
            thumb.Position = UDim2.new(1, 0, 0.5, 0)
            valLabel.Text = FormatVal(defVal)

            -- Dragging
            local sliding = false
            track.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1
                or input.UserInputType == Enum.UserInputType.Touch then
                    sliding = true
                    local sc = (input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X
                    UpdateSlider(sc)
                end
            end)
            InputSvc.InputChanged:Connect(function(input)
                if sliding and (input.UserInputType == Enum.UserInputType.MouseMovement
                    or input.UserInputType == Enum.UserInputType.Touch) then
                    local sc = (input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X
                    -- Update visuals without saving on every frame
                    sc = math.clamp(sc, 0, 1)
                    sliderVal = minVal + (maxVal - minVal) * sc
                    if decAmt == 0 then sliderVal = math.round(sliderVal) end
                    valLabel.Text = FormatVal(sliderVal)
                    fill.Size = UDim2.new(sc, 0, 1, 0)
                    thumb.Position = UDim2.new(1, 0, 0.5, 0)
                    pcall(cb, sliderVal)
                end
            end)
            InputSvc.InputEnded:Connect(function(input)
                if sliding and (input.UserInputType == Enum.UserInputType.MouseButton1
                    or input.UserInputType == Enum.UserInputType.Touch) then
                    sliding = false
                    -- Save only once when user releases
                    flags[key] = sliderVal
                    SaveFlags()
                end
            end)

            -- Hover thumb grow
            track.MouseEnter:Connect(function()
                Tween(thumb, 0.15, { Size = UDim2.fromOffset(17,17) })
            end)
            track.MouseLeave:Connect(function()
                if not sliding then
                    Tween(thumb, 0.2, { Size = UDim2.fromOffset(14,14) })
                end
            end)

            local SlidAPI = {}
            function SlidAPI:Set(v)
                local sc = (v - minVal) / (maxVal - minVal)
                UpdateSlider(sc)
            end
            function SlidAPI:Get() return sliderVal end
            return SlidAPI
        end

                    PaddingTop  = UDim.new(0,4), PaddingBottom = UDim.new(0,4),
                    Parent      = scroll,
                })

                Tween(popup, 0.2, { GroupTransparency = 0 }, Enum.EasingStyle.Quart)

                for _, opt in ipairs(options) do
                    local isSelected = (tostring(flags[key]) == tostring(opt))
                    local optBtn = New("TextButton", {
                        BackgroundColor3 = isSelected and LerpColor(T.Component, T.Accent, 0.18) or T.Component,
                        Size             = UDim2.new(1, 0, 0, 30),
                        Text             = "",

                        BackgroundTransparency = 1,

                        Font             = isSelected and Enum.Font.GothamBold or Enum.Font.Gotham,
                        TextSize         = 12,
                        TextXAlignment   = Enum.TextXAlignment.Left,
                        Size             = UDim2.new(1, -22, 1, 0),
                        Position         = UDim2.new(0, 10, 0, 0),
                        ZIndex           = 1003,
                        Parent           = optBtn,

                            BackgroundTransparency = 1,

                            ZIndex           = 1003,

                        Tween(popup, 0.18, { GroupTransparency = 1 }, Enum.EasingStyle.Quad)
                        task.delay(0.2, function() popup:Destroy() end)
                        SaveFlags()
                        pcall(cb, opt)
                    end)
                end

                -- Close on outside click
                local outsideConn
                outsideConn = InputSvc.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        local mp = Vector2.new(input.Position.X, input.Position.Y)
                        local pa = popup.AbsolutePosition
                        local ps = popup.AbsoluteSize
                        if mp.X < pa.X or mp.X > pa.X + ps.X or mp.Y < pa.Y or mp.Y > pa.Y + ps.Y then
                            ddOpen = false
                            Tween(chevron, 0.2, { Rotation = 0 })
                            Tween(popup, 0.18, { GroupTransparency = 1 }, Enum.EasingStyle.Quad)
                            task.delay(0.2, function() popup:Destroy() end)
                            outsideConn:Disconnect()
                        end
                    end
                end)
            end)

            local DdAPI = {}

                flags[key] = v
                selectedLabel.Text = tostring(v)
            end

            end

                PressEffect(b)
                AddRipple(b)
                return b
            end

            local card = New("Frame", {
                BackgroundColor3 = T.Component,
                Size             = UDim2.new(1, 0, 0, 62),
                LayoutOrder      = opts.Order or 0,
                Parent           = page,
            })
            Corner(10, card)
            Stroke(1, T.Outline, 0, card)

            New("TextLabel", {
                BackgroundTransparency = 1,
                Text             = opts.Title or "Theme",
                TextColor3       = T.Title,
                Font             = Enum.Font.GothamSemibold,
                TextSize         = 13,
                TextXAlignment   = Enum.TextXAlignment.Left,
                Size             = UDim2.new(1, -16, 0, 20),
                Position         = UDim2.new(0, 12, 0, 6),
                ZIndex           = card.ZIndex + 1,
                Parent           = card,
            })

            local dotRow = New("Frame", {
                BackgroundTransparency = 1,
                Size             = UDim2.new(1, -24, 0, 24),
                Position         = UDim2.new(0, 12, 1, -8),
                AnchorPoint      = Vector2.new(0, 1),
                ZIndex           = card.ZIndex + 1,
                Parent           = card,
            })
            New("UIListLayout", {
                FillDirection    = Enum.FillDirection.Horizontal,
                VerticalAlignment = Enum.VerticalAlignment.Center,
                Padding          = UDim.new(0, 8),
                Parent           = dotRow,
            })

            for _, tn in ipairs(themeNames) do
                local dotBtn = New("TextButton", {
                    BackgroundColor3 = accentColors[tn],
                    Size             = UDim2.fromOffset(20, 20),
                    Text             = "",
                    AutoButtonColor  = false,
                    BackgroundTransparency = (tn == Win.currentTheme) and 0 or 0.4,
                    ZIndex           = card.ZIndex + 2,
                    Parent           = dotRow,
                })
                Corner(100, dotBtn)
                Stroke(1.5, Color3.new(1,1,1),
                    (tn == Win.currentTheme) and 0.2 or 1, dotBtn)

                dotBtn.MouseEnter:Connect(function()
                    Tween(dotBtn, 0.14, { Size = UDim2.fromOffset(24,24), BackgroundTransparency = 0 })
                end)
                dotBtn.MouseLeave:Connect(function()
                    local active = tn == Win.currentTheme
                    Tween(dotBtn, 0.18, {
                        Size = UDim2.fromOffset(20, 20),
                        BackgroundTransparency = active and 0 or 0.4,
                    })
                end)
                dotBtn.MouseButton1Click:Connect(function()
                    Win.currentTheme = tn
                    ApplyTheme(tn)
                    winStroke.Color = T.Outline
                    Window.BackgroundColor3 = T.Primary
                    Sidebar.BackgroundColor3 = T.Sidebar
                    ContentArea.BackgroundColor3 = T.Secondary
                    TitleLabel.TextColor3 = T.Title
                    for n2, d2 in pairs(themeActiveDots) do
                        Tween(d2, 0.2, { BackgroundTransparency = (n2 == tn) and 0 or 0.35 })
                    end
                    Notify({ Title = "Theme", Description = tn .. " applied", Duration = 2, Type = "info" })
                end)
            end
        end

        -- Return tab page API
        return TabAPI
    end

    -- ──────────────────────────────────────────────────
    --  SET ACTIVE TAB
    -- ──────────────────────────────────────────────────
    function Win:SetTab(name)
        if currentTab == name then return end
        currentTab = name

        for tname, tdata in pairs(tabButtons) do
            local isActive = tname == name
            if isActive then
                Tween(tdata.btn,      0.2, { BackgroundColor3 = T.Component, BackgroundTransparency = 0.6 })
                Tween(tdata.activeBar, 0.2, { BackgroundTransparency = 0 })
                if tdata.titleText then
                    Tween(tdata.titleText, 0.2, { TextColor3 = T.Accent })
                    tdata.titleText.Font = Enum.Font.GothamBold
                end
                if tdata.iconImg then
                    Tween(tdata.iconImg, 0.2, { ImageColor3 = T.Accent })
                end
            else
                Tween(tdata.btn,      0.2, { BackgroundTransparency = 1 })
                Tween(tdata.activeBar, 0.2, { BackgroundTransparency = 1 })
                if tdata.titleText then
                    Tween(tdata.titleText, 0.2, { TextColor3 = T.Tab })
                    tdata.titleText.Font = Enum.Font.GothamSemibold
                end
                if tdata.iconImg then
                    Tween(tdata.iconImg, 0.2, { ImageColor3 = T.Icon })
                end
            end
        end

        for pname, pframe in pairs(tabPages) do
            if pname == name then
                pframe.Visible = true
                -- Fade-in via BackgroundTransparency on scrollbar (ScrollingFrame has no GroupTransparency)
                -- We just flip visibility; the sidebar active-bar provides transition feedback
            else
                pframe.Visible = false
            end
        end
    end

    -- ──────────────────────────────────────────────────
    --  SET TITLE
    -- ──────────────────────────────────────────────────
    function Win:SetTitle(t)
        TitleLabel.Text = t
    end

    -- ──────────────────────────────────────────────────
    --  NOTIFY (shortcut)
    -- ──────────────────────────────────────────────────
    function Win:Notify(opts) Notify(opts) end

    -- ──────────────────────────────────────────────────
    --  DESTROY
    -- ──────) ───────────────────────────────────────────
    function Win:Destroy()
        AnimateClose(Window, function()
            Window:Destroy()
            MobileBtn:Destroy()
        end)
    end

    -- Open animation
    AnimateOpen(Window, transp)

    return Win
end
