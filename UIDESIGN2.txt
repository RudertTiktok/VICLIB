--// Default fallback icon (a tiny dot) if name not found
local DEFAULT_ICON = "rbxassetid://7733919881"

local function GetIcon(Name)
    if not Name then return DEFAULT_ICON end
    -- Accept "home", "Home", "HOME", "home-2" etc
    local key = string.lower(tostring(Name))
    if Icons[key] then return Icons[key] end
    -- Try Title case → kebab (Home -> home)
    local alt = string.gsub(string.lower(key), " ", "-")
    if Icons[alt] then return Icons[alt] end
    return DEFAULT_ICON
end

--// ============================================================
--// UTILITY FUNCTIONS

local function Create(Class, Props)
    local Inst = Instance.new(Class)
    if Props then
        for k, v in next, Props do
            Inst[k] = v
        end
    end
    return Inst
end

local function Tween(Object, Duration, Properties, EasingStyle, EasingDirection)
    local Style = EasingStyle or Enum.EasingStyle.Quad
    local Direction = EasingDirection or Enum.EasingDirection.Out
    local Info = TweenInfo.new(Duration, Style, Direction)
    local t = TweenService:Create(Object, Info, Properties)
    t:Play()
    return t
end

--// Smooth color lerp
local function LerpColor(a, b, t)
    return Color3.fromRGB(
        math.floor(a.R * 255 + (b.R - a.R) * t * 255 + 0.5),
        math.floor(a.G * 255 + (b.G - a.G) * t * 255 + 0.5),
        math.floor(a.B * 255 + (b.B - a.B) * t * 255 + 0.5)
    )
end

--// Adjust color brightness (positive=lighten, negative=darken)
local function ShiftColor(color, amount)
    if not color then return Color3.fromRGB(128, 128, 128) end
    local r = math.clamp(math.floor((color.R or 0.5) * 255 + amount + 0.5), 0, 255)
    local g = math.clamp(math.floor((color.G or 0.5) * 255 + amount + 0.5), 0, 255)
    local b = math.clamp(math.floor((color.B or 0.5) * 255 + amount + 0.5), 0, 255)
    return Color3.fromRGB(r, g, b)
end

--// HSV → RGB
local function HSVtoColor(h, s, v)
    return Color3.fromHSV(h, s, v)
end

--// Get perceived brightness (0..1)
local function Brightness(color)
    return 0.299 * color.R + 0.587 * color.G + 0.114 * color.B
end

--// Choose black or white text based on background brightness
local function GetTextColor(bg)
    return Brightness(bg) > 0.55 and Color3.fromRGB(20, 20, 25) or Color3.fromRGB(255, 255, 255)
end

--// Multiply UDim2 by a scalar (used for size pulsing animations)
local function MultiplyUDim2(ud, factor)
    return UDim2.new(ud.X.Scale * factor, ud.X.Offset * factor, ud.Y.Scale * factor, ud.Y.Offset * factor)
end

--// Format number with optional decimals
local function FormatNumber(n, decimals)
    if decimals and decimals > 0 then
        local p = 10 ^ decimals
        return tostring(math.floor(n * p + 0.5) / p)
    end
    return tostring(math.floor(n + 0.5))
end

--// ============================================================
--// SAFE PARENTING - gethui() first, then CoreGui fallback
--// ============================================================
local function GetSafeParent()
    if gethui then
        return gethui()
    elseif syn and syn.protect_gui then
        return CoreGui
    elseif protectgui then
        return CoreGui
    else
        -- Try CoreGui first, fall back to PlayerGui
        local ok = pcall(function()
            return CoreGui:FindFirstChild("RobloxGui")
        end)
        if ok then return CoreGui end
    end
    return LocalPlayer:WaitForChild("PlayerGui")
end

local function SafeParent(gui)
    local parent = GetSafeParent()
    if gethui then
        gui.Parent = parent
    elseif syn and syn.protect_gui then
        syn.protect_gui(gui)
        gui.Parent = parent
    elseif protectgui then
        protectgui(gui)
        gui.Parent = parent
    else
        local ok = pcall(function()
            gui.Parent = parent
        end)
        if not ok then
            gui.Parent = LocalPlayer:WaitForChild("PlayerGui")
        end
    end
end

--// ============================================================
--// DRAGGABLE - smooth drag with InputBegan/InputChanged pattern
--// ============================================================
local function MakeDraggable(frame, dragArea)
    dragArea = dragArea or frame
    local dragging = false
    local dragStart, startPos

    dragArea.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    dragArea.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            if dragging then
                local delta = input.Position - dragStart
                frame.Position = UDim2.new(
                    startPos.X.Scale, startPos.X.Offset + delta.X,
                    startPos.Y.Scale, startPos.Y.Offset + delta.Y
                )
            end
        end
    end)
end

--// ============================================================
--// PREMIUM HOVER ANIMATION SYSTEM
--// ============================================================
local function ApplyHover(frame, opt)
    opt = opt or {}
    local theme = opt.Theme
    local onEnter = opt.OnEnter
    local onLeave = opt.OnLeave
    local scaleFactor = opt.Scale or 1.0
    local bgLerp = opt.Lerp or 0.15
    local targetColor = opt.TargetColor
    local originalColor = frame.BackgroundColor3

    frame.MouseEnter:Connect(function()
        if onEnter then onEnter() end
        if targetColor then
            Tween(frame, 0.2, {
                BackgroundColor3 = LerpColor(originalColor, targetColor, bgLerp),
            }, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
        end
        if scaleFactor ~= 1.0 then
            local orig = frame.Size
            Tween(frame, 0.15, {
                Size = UDim2.new(orig.X.Scale, orig.X.Offset * scaleFactor, orig.Y.Scale, orig.Y.Offset * scaleFactor),
            }, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
        end
    end)

    frame.MouseLeave:Connect(function()
        if onLeave then onLeave() end
        if targetColor then
            Tween(frame, 0.25, {
                BackgroundColor3 = originalColor,
            }, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
        end
        if scaleFactor ~= 1.0 then
            local orig = frame.Size
            -- Get back to original scale via storing once on enter
            Tween(frame, 0.25, {
                BackgroundColor3 = originalColor,
            })
        end
    end)
end

--// Hover for buttons - simpler, more reliable
local function HoverButton(btn, opt)
    opt = opt or {}
    local originalColor = btn.BackgroundColor3 or Color3.fromRGB(60, 60, 70)
    local hoverColor = opt.HoverColor or ShiftColor(originalColor, 12)
    local pressColor = opt.PressColor or ShiftColor(originalColor, -8)
    local onEnter = opt.OnEnter
    local onLeave = opt.OnLeave

    if btn.MouseEnter then
        btn.MouseEnter:Connect(function()
            if onEnter then onEnter() end
            Tween(btn, 0.18, {
                BackgroundColor3 = hoverColor,
            }, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
        end)
    end
    if btn.MouseLeave then
        btn.MouseLeave:Connect(function()
            if onLeave then onLeave() end
            Tween(btn, 0.22, {
                BackgroundColor3 = originalColor,
            }, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
        end)
    end
    if btn.MouseButton1Down then
        btn.MouseButton1Down:Connect(function()
            Tween(btn, 0.08, {
                BackgroundColor3 = pressColor,
            }, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
        end)
    end
    if btn.MouseButton1Up then
        btn.MouseButton1Up:Connect(function()
            Tween(btn, 0.18, {
                BackgroundColor3 = hoverColor,
            }, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
        end)
    end
end

--// Add an instance to the theme registry so its colors auto-update on theme change
function Library:AddToRegistry(instance, properties)
    table.insert(self.Registry, {Instance = instance, Properties = properties})
    -- Apply immediately
    for prop, key in next, properties do
        if Library.Theme[key] then
            instance[prop] = Library.Theme[key]
        end
    end
    return instance
end

--// Apply current theme to all registered instances
function Library:ApplyTheme()
    for _, entry in ipairs(self.Registry) do
        for prop, key in next, entry.Properties do
            if Library.Theme[key] then
                local inst = entry.Instance
                if inst and inst.Parent then
                    Tween(inst, 0.25, {[prop] = Library.Theme[key]}, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
                end
            end
        end
    end
end

        --// AddSlider
        function TabOptions:AddSlider(opts)
            opts = opts or {}
            local minVal = opts.MinValue or 0
            local maxVal = opts.MaxValue or 100
            local defaultVal = opts.Default or minVal
            if opts.Flag and Library.Flags[opts.Flag] ~= nil then
                defaultVal = Library.Flags[opts.Flag]
            end
            local decimals = opts.Decimals or 0
            local step = opts.Step or (decimals > 0 and 1 / (10 ^ decimals) or 1)
            local value = math.clamp(defaultVal, minVal, maxVal)

            local card, stroke = BuildCard(tabContent, Library.Theme)
            -- Top: title + value
            local header = Create("Frame", {
                Parent = card,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 0, 22),
            })
            local titleLabel = Create("TextLabel", {
                Parent = header,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, -60, 1, 0),
                FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold),
                TextColor3 = Library.Theme.Title,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
                Text = opts.Title or "Slider",
            })
            local valueLabel = Create("TextLabel", {
                Parent = header,
                BackgroundTransparency = 1,
                Position = UDim2.new(1, -60, 0, 0),
                Size = UDim2.fromOffset(60, 22),
                FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Bold),
                TextColor3 = Library.Theme.Accent,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Right,
                Text = tostring(value),
            })
            Library:AddToRegistry(titleLabel, {TextColor3 = "Title"})
            Library:AddToRegistry(valueLabel, {TextColor3 = "Accent"})

            -- Description below
            if opts.Description then
                local d = Create("TextLabel", {
                    Parent = card,
                    BackgroundTransparency = 1,
                    Size = UDim2.new(1, 0, 0, 14),
                    FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Regular),
                    TextColor3 = Library.Theme.Subtext,
                    TextSize = 11,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    Text = opts.Description,
                })
                Library:AddToRegistry(d, {TextColor3 = "Subtext"})
            end

            -- Slider track
            local track = Create("Frame", {
                Name = "Track",
                Parent = card,
                BackgroundColor3 = Library.Theme.Interactable,
                BorderSizePixel = 0,
                Size = UDim2.new(1, 0, 0, 8),
                Position = UDim2.fromOffset(0, 38),
            })
            Create("UICorner", {Parent = track, CornerRadius = UDim.new(1, 0)})
            local trackStroke = Create("UIStroke", {Parent = track, Color = Library.Theme.Stroke, Thickness = 1, Transparency = 0.5})
            Library:AddToRegistry(track, {BackgroundColor3 = "Interactable"})
            Library:AddToRegistry(trackStroke, {Color = "Stroke"})

            -- Fill
            local fill = Create("Frame", {
                Name = "Fill",
                Parent = track,
                BackgroundColor3 = Library.Theme.Accent,
                BorderSizePixel = 0,
                Size = UDim2.fromScale((value - minVal) / (maxVal - minVal), 1),
            })
            Create("UICorner", {Parent = fill, CornerRadius = UDim.new(1, 0)})
            -- Gradient on fill
            local fillGrad = Create("UIGradient", {
                Parent = fill,
                Color = ColorSequence.new({
                    ColorSequenceKeypoint.new(0, Library.Theme.AccentSoft),
                    ColorSequenceKeypoint.new(1, Library.Theme.Accent),
                }),
            })
            Library:AddToRegistry(fill, {BackgroundColor3 = "Accent"})

            -- Knob
            local knob = Create("Frame", {
                Name = "Knob",
                Parent = track,
                BackgroundColor3 = Color3.fromRGB(255, 255, 255),
                BorderSizePixel = 0,
                AnchorPoint = Vector2.new(0.5, 0.5),
                Position = UDim2.new((value - minVal) / (maxVal - minVal), 0, 0.5, 0),
                Size = UDim2.fromOffset(18, 18),
                ZIndex = 3,
            })
            Create("UICorner", {Parent = knob, CornerRadius = UDim.new(1, 0)})
            local knobStroke = Create("UIStroke", {Parent = knob, Color = Library.Theme.Accent, Thickness = 2, Transparency = 0})
            Library:AddToRegistry(knobStroke, {Color = "Accent"})

            -- Make card auto-size to fit slider
            card.Size = UDim2.new(1, 0, 0, 70)

            local dragging = false
            local function updateFromMouse()
                local relX = math.clamp((Mouse.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
                local raw = minVal + relX * (maxVal - minVal)
                -- Snap to step
                raw = math.floor(raw / step + 0.5) * step
                value = math.clamp(raw, minVal, maxVal)
                if decimals > 0 then
                    valueLabel.Text = string.format("%." .. decimals .. "f", value)
                else
                    valueLabel.Text = tostring(math.floor(value + 0.5))
                end
                local scale = (value - minVal) / (maxVal - minVal)
                Tween(fill, 0.05, {Size = UDim2.fromScale(scale, 1)}, Enum.EasingStyle.Linear)
                Tween(knob, 0.05, {Position = UDim2.new(scale, 0, 0.5, 0)}, Enum.EasingStyle.Linear)
                if opts.Flag then
                    Library.Flags[opts.Flag] = value
                    Library:SaveFlags()
                end
                if opts.Callback then opts.Callback(value) end
            end

            track.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                    dragging = true
                    updateFromMouse()
                end
            end)
            UserInputService.InputEnded:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                    dragging = false
                end
            end)
            UserInputService.InputChanged:Connect(function(input)
                if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                    updateFromMouse()
                end
            end)
            -- Knob hover scale
            knob.MouseEnter:Connect(function()
                Tween(knob, 0.15, {Size = UDim2.fromOffset(22, 22)}, Enum.EasingStyle.Back, Enum.EasingDirection.Out)
            end)
            knob.MouseLeave:Connect(function()
                Tween(knob, 0.2, {Size = UDim2.fromOffset(18, 18)})
            end)

            return {
                Set = function(v)
                    value = math.clamp(v, minVal, maxVal)
                    local scale = (value - minVal) / (maxVal - minVal)
                    if decimals > 0 then
                        valueLabel.Text = string.format("%." .. decimals .. "f", value)
                    else
                        valueLabel.Text = tostring(math.floor(value + 0.5))
                    end
                    Tween(fill, 0.2, {Size = UDim2.fromScale(scale, 1)})
                    Tween(knob, 0.2, {Position = UDim2.new(scale, 0, 0.5, 0)})
                    if opts.Callback then opts.Callback(value) end
                end,
                Get = function() return value end,
            }
        end

                AnchorPoint = Vector2.new(1, 0.5),

                Size = UDim2.fromOffset(16, 16),
                Image = GetIcon("chevron-down"),
                ImageColor3 = Library.Theme.Subtext,
            })
            Library:AddToRegistry(chev, {ImageColor3 = "Subtext"})

            -- Dropdown panel (modal)
            local function closeDropdown(panel)
                Tween(panel, 0.2, {Size = UDim2.new(1, 0, 0, 0)}, Enum.EasingStyle.Quad, Enum.EasingDirection.In)
                Tween(chev, 0.2, {Rotation = 0})
                task.delay(0.2, function()
                    if panel and panel.Parent then panel:Destroy() end
                end)
                expanded = false
            end

            local function openDropdown()
                if expanded then return end
                expanded = true
                Tween(chev, 0.2, {Rotation = 180})
                local absPos = valueContainer.AbsolutePosition
                local absSize = valueContainer.AbsoluteSize
                local panel = Create("Frame", {
                    Name = "DropdownPanel",
                    Parent = ScreenGui,
                    BackgroundColor3 = Library.Theme.Card,
                    BorderSizePixel = 0,
                    Position = UDim2.fromOffset(absPos.X, absPos.Y + absSize.Y + 4),
                    Size = UDim2.fromOffset(absSize.X, 0),
                    ClipsDescendants = true,
                    ZIndex = 200,
                })
                Create("UICorner", {Parent = panel, CornerRadius = UDim.new(0, 8)})
                local pStroke = Create("UIStroke", {Parent = panel, Color = Library.Theme.Stroke, Thickness = 1, Transparency = 0})
                Library:AddToRegistry(panel, {BackgroundColor3 = "Card"})
                Library:AddToRegistry(pStroke, {Color = "Stroke"})
                Create("UIListLayout", {
                    Parent = panel,
                    FillDirection = Enum.FillDirection.Vertical,
                    Padding = UDim.new(0, 2),
                    SortOrder = Enum.SortOrder.LayoutOrder,
                })
                Create("UIPadding", {Parent = panel, PaddingLeft = UDim.new(0,4), PaddingRight = UDim.new(0,4), PaddingTop = UDim.new(0,4), PaddingBottom = UDim.new(0,4)})
                for i, opt in ipairs(options) do
                    local item = Create("TextButton", {
                        Parent = panel,
                        BackgroundColor3 = opt == selected and Library.Theme.AccentSoft or Library.Theme.Component,
                        BackgroundTransparency = opt == selected and 0 or 0.4,
                        BorderSizePixel = 0,
                        Size = UDim2.new(1, 0, 0, 30),
                        AutoButtonColor = false,
                        Text = "",
                        LayoutOrder = i,
                    })
                    Create("UICorner", {Parent = item, CornerRadius = UDim.new(0, 6)})
                    local itemLabel = Create("TextLabel", {
                        Parent = item,
                        BackgroundTransparency = 1,
                        Position = UDim2.fromOffset(10, 0),
                        Size = UDim2.new(1, -20, 1, 0),
                        FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Medium),
                        TextColor3 = opt == selected and Library.Theme.AccentText or Library.Theme.Title,
                        TextSize = 13,
                        TextXAlignment = Enum.TextXAlignment.Left,
                        Text = tostring(opt),
                    })
                    HoverButton(item, {HoverColor = Library.Theme.CardHover})
                    item.MouseButton1Click:Connect(function()
                        selected = opt
                        valueLabel.Text = tostring(opt)
                        if opts.Flag then
                            Library.Flags[opts.Flag] = opt
                            Library:SaveFlags()
                        end
                        if opts.Callback then opts.Callback(opt) end
                        closeDropdown(panel)
                    end)
                end
                -- Animate open
                panel.Size = UDim2.fromOffset(absSize.X, 0)
                Tween(panel, 0.25, {Size = UDim2.fromOffset(absSize.X, math.min(#options * 34 + 8, 220))}, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out)
                -- Click outside to close
                local onClose
                onClose = UserInputService.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        local mPos = UserInputService:GetMouseLocation()
                        local pPos = panel.AbsolutePosition
                        local pSize = panel.AbsoluteSize
                        if mPos.X < pPos.X or mPos.X > pPos.X + pSize.X or mPos.Y < pPos.Y or mPos.Y > pPos.Y + pSize.Y then
                            closeDropdown(panel)
                            onClose:Disconnect()
                        end
                    end
                end)
            end

            valueContainer.MouseButton1Click:Connect(function()
                if expanded then
                    -- close current - find panel
                    local panel = ScreenGui:FindFirstChild("DropdownPanel")
                    if panel then closeDropdown(panel) end
                else
                    openDropdown()
                end
            end)
            return {
                Set = function(v)
                    selected = v
                    valueLabel.Text = tostring(v)
                    if opts.Callback then opts.Callback(v) end
                end,
                Get = function() return selected end,
                SetOptions = function(newOpts)
                    options = newOpts
                end,
            }
        end

            opts = opts or {}
            local currentBind = opts.Default

            local listening = false

            Library:AddToRegistry(bStroke, {Color = "Stroke"})
            local bindLabel = Create("TextLabel", {
                Parent = bindBtn,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 1, 0),
                FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Medium),
                TextColor3 = Library.Theme.Title,
                TextSize = 13,
                Text = currentBind and tostring(currentBind.Name or currentBind):gsub("Enum.KeyCode.", "") or "None",
            })
            Library:AddToRegistry(bindLabel, {TextColor3 = "Title"})

            local function setListening(v)
                listening = v
                if v then
                    bindLabel.Text = "Press key..."
                    Tween(bindBtn, 0.2, {BackgroundColor3 = Library.Theme.AccentSoft}, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
                    Tween(bStroke, 0.2, {Color = Library.Theme.Accent, Transparency = 0})
                else
                    Tween(bindBtn, 0.2, {BackgroundColor3 = Library.Theme.Interactable}, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
                    Tween(bStroke, 0.2, {Color = Library.Theme.Stroke, Transparency = 0.5})
                end
            end

            bindBtn.MouseButton1Click:Connect(function()
                setListening(not listening)
            end)

            UserInputService.InputBegan:Connect(function(input, processed)
                if listening then
                    if input.UserInputType == Enum.UserInputType.Keyboard then
                        currentBind = input.KeyCode
                        bindLabel.Text = tostring(input.KeyCode.Name)
                        if opts.Flag then
                            Library.Flags[opts.Flag] = input.KeyCode.Name
                            Library:SaveFlags()
                        end
                        setListening(false)
                        if opts.Callback then opts.Callback(input.KeyCode) end
                    elseif input.UserInputType == Enum.UserInputType.MouseButton1 then
                        -- cancel
                        setListening(false)
                    end
                elseif not processed and currentBind and input.KeyCode == currentBind then
                    if opts.Callback then opts.Callback(input.KeyCode) end
                end
            end)

            return {
                Set = function(key)
                    currentBind = key
                    bindLabel.Text = tostring(key.Name or key)
                end,
                Get = function() return currentBind end,
            }
        end

                        ColorSequenceKeypoint.new(0, Color3.fromRGB(255, 255, 255)),
                        ColorSequenceKeypoint.new(1, Color3.fromRGB(0, 0, 0)),
                    }),
                    Transparency = NumberSequence.new({
                        NumberSequenceKeypoint.new(0, 1),
                        NumberSequenceKeypoint.new(1, 0),
                    }),
                })
                local svGrad2 = Create("UIGradient", {
                    Parent = svPicker,
                    Color = ColorSequence.new({
                        ColorSequenceKeypoint.new(0, Color3.fromRGB(255, 255, 255)),
                        ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 255, 255)),
                    }),
                    Transparency = NumberSequence.new({
                        NumberSequenceKeypoint.new(0, 0),
                        NumberSequenceKeypoint.new(1, 1),
                    }),
                    Rotation = 90,
                })
                local svCursor = Create("Frame", {
                    Parent = svPicker,
                    BackgroundColor3 = Color3.fromRGB(255, 255, 255),
                    BorderSizePixel = 0,
                    AnchorPoint = Vector2.new(0.5, 0.5),
                    Size = UDim2.fromOffset(8, 8),
                    Position = UDim2.new(s, 0, 1 - v, 0),
                    ZIndex = 5,
                })
                Create("UICorner", {Parent = svCursor, CornerRadius = UDim.new(1, 0)})
                Create("UIStroke", {Parent = svCursor, Color = Color3.fromRGB(0,0,0), Thickness = 1, Transparency = 0})

                -- Hue strip
                local huePicker = Create("TextButton", {
                    Name = "HuePicker",
                    Parent = panel,
                    BackgroundColor3 = Color3.fromRGB(255, 255, 255),
                    BorderSizePixel = 0,
                    Position = UDim2.new(0, 0, 0, 150),
                    Size = UDim2.new(1, 0, 0, 14),
                    AutoButtonColor = false,
                    Text = "",
                })
                Create("UICorner", {Parent = huePicker, CornerRadius = UDim.new(1, 0)})
                Create("UIGradient", {
                    Parent = huePicker,
                    Color = ColorSequence.new({
                        ColorSequenceKeypoint.new(0, Color3.fromRGB(255, 0, 0)),
                        ColorSequenceKeypoint.new(0.17, Color3.fromRGB(255, 255, 0)),
                        ColorSequenceKeypoint.new(0.33, Color3.fromRGB(0, 255, 0)),
                        ColorSequenceKeypoint.new(0.5, Color3.fromRGB(0, 255, 255)),
                        ColorSequenceKeypoint.new(0.67, Color3.fromRGB(0, 0, 255)),
                        ColorSequenceKeypoint.new(0.83, Color3.fromRGB(255, 0, 255)),
                        ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 0, 0)),
                    }),
                })
                local hueCursor = Create("Frame", {
                    Parent = huePicker,
                    BackgroundColor3 = Color3.fromRGB(255, 255, 255),
                    BorderSizePixel = 0,
                    AnchorPoint = Vector2.new(0.5, 0.5),
                    Size = UDim2.fromOffset(6, 18),
                    Position = UDim2.new(h, 0, 0.5, 0),
                    ZIndex = 5,
                })
                Create("UICorner", {Parent = hueCursor, CornerRadius = UDim.new(1, 0)})
                Create("UIStroke", {Parent = hueCursor, Color = Color3.fromRGB(0,0,0), Thickness = 1, Transparency = 0})

                local svDragging = false
                local hueDragging = false

                local function updateSV()
                    local relX = math.clamp((Mouse.X - svPicker.AbsolutePosition.X) / svPicker.AbsoluteSize.X, 0, 1)
                    local relY = math.clamp((Mouse.Y - svPicker.AbsolutePosition.Y) / svPicker.AbsoluteSize.Y, 0, 1)
                    s = relX
                    v = 1 - relY
                    svCursor.Position = UDim2.new(s, 0, 1 - v, 0)
                    svPicker.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
                    updateColor(h, s, v)
                end

                local function updateHue()
                    local relX = math.clamp((Mouse.X - huePicker.AbsolutePosition.X) / huePicker.AbsoluteSize.X, 0, 1)
                    h = relX
                    hueCursor.Position = UDim2.new(h, 0, 0.5, 0)
                    svPicker.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
                    updateColor(h, s, v)
                end

                svPicker.MouseButton1Down:Connect(function()
                    svDragging = true
                    updateSV()
                end)
                huePicker.MouseButton1Down:Connect(function()
                    hueDragging = true
                    updateHue()
                end)
                local moveConn
                moveConn = UserInputService.InputChanged:Connect(function(input)
                    if svDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                        updateSV()
                    elseif hueDragging then
                        updateHue()
                    end
                end)
                UserInputService.InputEnded:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        svDragging = false
                        hueDragging = false
                    end
                end)

                -- Animate open
                panel.Size = UDim2.fromOffset(220, 0)
                Tween(panel, 0.25, {Size = UDim2.fromOffset(220, 184)}, Enum.EasingStyle.Exponential, Enum.EasingDirection.Out)

                -- Click outside to close
                local closeConn
                closeConn = UserInputService.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        local mPos = UserInputService:GetMouseLocation()
                        local pPos = panel.AbsolutePosition
                        local pSize = panel.AbsoluteSize
                        if mPos.X < pPos.X or mPos.X > pPos.X + pSize.X or mPos.Y < pPos.Y or mPos.Y > pPos.Y + pSize.Y then
                            closePicker(panel)
                            closeConn:Disconnect()
                            moveConn:Disconnect()
                        end
                    end
                end)
            end

            swatch.MouseButton1Click:Connect(function()
                if pickerOpen then
                    local panel = ScreenGui:FindFirstChild("ColorPickerPanel")
                    if panel then closePicker(panel) end
                else
                    openPicker()
                end
            end)
            HoverButton(swatch, {HoverColor = ShiftColor(currentColor, 20)})

            return {
                Set = function(c)
                    currentColor = c
                    h, s, v = Color3.toHSV(c)
                    swatch.BackgroundColor3 = c
                    if opts.Callback then opts.Callback(c) end
                end,
                Get = function() return currentColor end,
            }
        end

        --// AddSegmented - segmented control
        function TabOptions:AddSegmented(opts)
            opts = opts or {}
            local options = opts.Options or {"A", "B", "C"}
            local selected = opts.Default or options[1]
            local card, stroke = BuildCard(tabContent, Library.Theme)
            local labels, titleLabel, descLabel = BuildLabels(card, opts.Title, opts.Description)

        --// AddStepper - numeric +/- stepper
        function TabOptions:AddStepper(opts)
            opts = opts or {}
            local minVal = opts.MinValue or 0
            local maxVal = opts.MaxValue or 100
            local step = opts.Step or 1
            local value = opts.Default or minVal
            if opts.Flag and Library.Flags[opts.Flag] ~= nil then
                value = Library.Flags[opts.Flag]
            end
            value = math.clamp(value, minVal, maxVal)

            local card, stroke = BuildCard(tabContent, Library.Theme)
            local labels, titleLabel, descLabel = BuildLabels(card, opts.Title, opts.Description)
            -- Stepper on right
            local stepper = Create("Frame", {
                Parent = card,
                BackgroundColor3 = Library.Theme.Interactable,
                BorderSizePixel = 0,
                AnchorPoint = Vector2.new(1, 0.5),

                Size = UDim2.fromOffset(120, 32),
            })
            Create("UICorner", {Parent = stepper, CornerRadius = UDim.new(0, 8)})
            Library:AddToRegistry(stepper, {BackgroundColor3 = "Interactable"})

            local minusBtn = Create("TextButton", {
                Parent = stepper,
                BackgroundTransparency = 1,
                BorderSizePixel = 0,
                Size = UDim2.fromOffset(32, 32),
                AutoButtonColor = false,
                Text = "",
            })
            local minusIcon = Create("ImageLabel", {
                Parent = minusBtn,
                BackgroundTransparency = 1,
                Position = UDim2.new(0.5, -10, 0.5, -10),
                Size = UDim2.fromOffset(20, 20),
                Image = GetIcon("minus"),
                ImageColor3 = Library.Theme.Title,
            })
            Library:AddToRegistry(minusIcon, {ImageColor3 = "Title"})

            local valueLabel = Create("TextLabel", {
                Parent = stepper,
                BackgroundTransparency = 1,
                Position = UDim2.fromOffset(32, 0),
                Size = UDim2.new(1, -64, 1, 0),
                FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold),
                TextColor3 = Library.Theme.Title,
                TextSize = 14,
                Text = tostring(value),
            })
            Library:AddToRegistry(valueLabel, {TextColor3 = "Title"})

            local plusBtn = Create("TextButton", {
                Parent = stepper,
                BackgroundTransparency = 1,
                BorderSizePixel = 0,
                Position = UDim2.new(1, -32, 0, 0),
                Size = UDim2.fromOffset(32, 32),
                AutoButtonColor = false,
                Text = "",
            })
            local plusIcon = Create("ImageLabel", {
                Parent = plusBtn,
                BackgroundTransparency = 1,
                Position = UDim2.new(0.5, -10, 0.5, -10),
                Size = UDim2.fromOffset(20, 20),
                Image = GetIcon("plus"),
                ImageColor3 = Library.Theme.Title,
            })
            Library:AddToRegistry(plusIcon, {ImageColor3 = "Title"})

            local function update(delta)
                value = math.clamp(value + delta * step, minVal, maxVal)
                valueLabel.Text = tostring(value)
                if opts.Flag then
                    Library.Flags[opts.Flag] = value
                    Library:SaveFlags()
                end
                if opts.Callback then opts.Callback(value) end
            end

            HoverButton(minusBtn, {HoverColor = Library.Theme.CardHover})
            HoverButton(plusBtn, {HoverColor = Library.Theme.CardHover})
            minusBtn.MouseButton1Click:Connect(function() update(-1) end)
            plusBtn.MouseButton1Click:Connect(function() update(1) end)
            return {
                Set = function(v) value = math.clamp(v, minVal, maxVal) valueLabel.Text = tostring(value) end,
                Get = function() return value end,
            }
        end

    function Options:SetTheme(name)
        if Themes[name] then
            Library.Theme = Themes[name]
            Library.ThemeName = name
            Library:ApplyTheme()
        end
    end

        return Library.ThemeName
    end

    function Options:GetThemes()
        local names = {}
        for k, _ in pairs(Themes) do table.insert(names, k) end

        end
    end

    function Options:Show()
        if not windowState.Visible then openWindow() end
    end

    function Options:Hide()
        if windowState.Visible then closeWindow() end
    end

    -- Initialize: open window
    openWindow()

    return Options
end

--// ============================================================
--// GLOBAL LIBRARY API
--// ============================================================
Library.Icons = Icons
Library.Themes = Themes
Library.GetIcon = GetIcon

function Library:GetIcon(name)
    return GetIcon(name)
end

function Library:GetTheme(name)
    return Themes[name]
end

function Library:ListThemes()
    local names = {}
    for k, _ in pairs(Themes) do table.insert(names, k) end
    table.sort(names)
    return names
end

if gethui then
    pcall(function()
        for _, child in ipairs(gethui():GetChildren()) do
            if child:IsA("ScreenGui") and child.Name:match("^VICLIB_") and child ~= ScreenGui then
                child:Destroy()
            end
        end
    end)
end

return Library
